\hspace{-6mm}\framebox[3ex]{\rule{0ex}{.7em}$1$}\quad%% 大問の番号
\begin{\ctoi}
$3$次の正方行列$A=
\left(
\begin{array}{rrr}
a_1 & b_1 & c_1\\
a_2 & b_2 & c_2\\
a_3 & b_3 & c_3\\
\end{array}
\right)
$の行列式$|A|$の値を$k(\neq 0)$とするとき，次の行列の値を答えよ。
\begin{edaenumerate}<3>[(1)\ ]%% 小問は，edaenumerateで
\item\ $\left|
\begin{array}{rrr}
a_2 & b_2 & c_2\\
a_1 & b_1 & c_1\\
a_3 & b_3 & c_3\\
\end{array}
\right|$
\item\ $\left|
\begin{array}{rrr}
 a_1 & b_1 & c_1\\
5a_2 &5b_2 &5c_2\\
 a_3 & b_3 & c_3\\
\end{array}
\right|$
\item\ $\left|
\begin{array}{rrr}
5a_2 &5b_2 &5c_2\\
5a_1 &5b_1 &5c_1\\
5a_3 &5b_3 &5c_3\\
\end{array}
\right|$
\end{edaenumerate}
\end{\ctoi}
%%
\begin{\ckai}(解答)\par
\small
\begin{edaenumerate}<1>[(1)\ ]%% 小問は，edaenumerateで
\item\ 第$1$列と第$2$列が交換されているから，$-k$
\item\ 第$2$行が$5$倍されているから，$5k$
\item\ 第$1$列と第$2$列が交換され,かつ,$3$つの列が$5$倍されているから，$-5^3k=-125k$
\end{edaenumerate}
\end{\ckai}

\hspace{-6mm}\framebox[3ex]{\rule{0ex}{.7em}$2$}\quad%% 大問の番号
\begin{\ctoi}
$3$次の正方行列について，$|-2A|=k|A|$をみたす定数$k$の値を答えよ。\\
ただし，$|A|\neq 0$とする。
\end{\ctoi}
%
\begin{\ckai}(解答)\par
$(-2)^3=-8$
\end{\ckai}

\hspace{-6mm}\framebox[3ex]{\rule{0ex}{.7em}$3$}\quad%% 大問の番号

$\left|
\begin{array}{rrr}

4 & 5 & 6\\
7 & 8 & 9\\
\end{array}
\right|
=a
\left|
\begin{array}{rr}
4 & 6 \\
7 & 9 \\
\end{array}
\right|
+b
\left|
\begin{array}{rr}
1 & 3 \\
7 & 9 \\
\end{array}
\right|
+c
\left|
\begin{array}{rr}
1 & 3 \\
4 & 6 \\
\end{array}
\right|$
\begin{edaenumerate}<3>[(1)\ ]%% 小問は，edaenumerateで
\item\ $a$の値を答えよ。 \item\ $b$の値を答えよ。 \item\ $c$の値を答えよ。
\end{edaenumerate}
\end{\ctoi}
%%
\begin{\ckai}(解答) ※ 符号に注意して答える。\par

\item\ $a=-2$ \item\ $b=5$ \item\ $c=-8$
\end{edaenumerate}
\end{\ckai}

\hspace{-6mm}\framebox[3ex]{\rule{0ex}{.7em}$5$}\quad%% 大問の番号
\begin{\ctoi}
$3$次の正方行列$A=
\left(
\begin{array}{rrr}
3 & 2 & 1\\
6 & 5 & 4\\
0 & 8 & 7\\
\end{array}
\right)
$について次のものを答えよ。
\begin{edaenumerate}<3>[(1)\ ]%% 小問は，edaenumerateで
\item\ $(1,\!1)\!$小行列式$D_{1\!1}$\par と$(1,\!1)$余因子$\tilde{a}_{1\!1}$
\item\ $(1,\!2)\!$小行列式$D_{1\!2}$\par と$(1,\!2)$余因子$\tilde{a}_{1\!2}$
\item\ $(1,\!3)\!$小行列式$D_{1\!3}$\par と$(1,\!3)$余因子$\tilde{a}_{1\!3}$
\item\ $(2,\!1)\!$小行列式$D_{2\!1}$\par と$(2,\!1)$余因子$\tilde{a}_{2\!1}$
\item\ $(2,\!2)\!$小行列式$D_{2\!2}$\par と$(2,\!2)$余因子$\tilde{a}_{2\!2}$
\item\ $(2,\!3)\!$小行列式$D_{2\!3}$\par と$(2,\!3)$余因子$\tilde{a}_{2\!3}$
\item\ $(3,\!1)\!$小行列式$D_{3\!1}$\par と$(3,\!1)$余因子$\tilde{a}_{3\!1}$
\item\ $(3,\!2)\!$小行列式$D_{3\!2}$\par と$(3,\!2)$余因子$\tilde{a}_{3\!2}$
\item\ $(3,\!3)\!$小行列式$D_{3\!3}$\par と$(3,\!3)$余因子$\tilde{a}_{3\!3}$
\end{edaenumerate}
\end{\ctoi}
%%
\begin{\ckai}(解答)\par
\begin{edaenumerate}<3>[(1)\ ]%% 小問は，edaenumerateで
\item\ $D_{11}=\left|\begin{array}{rr}5&4\\ 8&7\end{array}\right|\\ \ =3$,
 \ $\tilde{a}_{11}=3$
\item\ $D_{12}=\left|\begin{array}{rr}6&4\\ 0&7\end{array}\right|\\ \ =42$,
 \ $\tilde{a}_{12}=-42$
\item\ $D_{13}=\left|\begin{array}{rr}6&5\\ 0&8\end{array}\right|\\ \ =48$,
 \ $\tilde{a}_{13}=48$
\item\ $D_{21}=\left|\begin{array}{rr}2&1\\ 8&7\end{array}\right|\\ \ =6$,
 \ $\tilde{a}_{21}=-6$
\item\ $D_{22}=\left|\begin{array}{rr}3&1\\ 0&7\end{array}\right|\\ \ =21$,
 \ $\tilde{a}_{22}=21$
\item\ $D_{23}=\left|\begin{array}{rr}3&2\\ 0&8\end{array}\right|\\ \ =24$,
 \ $\tilde{a}_{23}=-24$
\item\ $D_{31}=\left|\begin{array}{rr}2&1\\ 5&4\end{array}\right|\\ \ =3$,
 \ $\tilde{a}_{31}=3$
\item\ $D_{32}=\left|\begin{array}{rr}3&1\\ 6&4\end{array}\right|\\ \ =6$,
 \ $\tilde{a}_{32}=-6$
\item\ $D_{33}=\left|\begin{array}{rr}3&2\\ 6&5\end{array}\right|\\ \ =3$,
 \ $\tilde{a}_{33}=3$
\end{edaenumerate}
\end{\ckai}
