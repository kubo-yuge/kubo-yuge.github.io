\hspace{-6mm}\framebox[3ex]{\rule{0ex}{.7em}$2$}\quad%% 大問の番号
\begin{\ctoi}
次の等式をみたす行列$X$を求めよ。
%% 実教出版「新版 線形代数」p.79練習20を利用する。
(授業中の課題「逆行列の利用（2次）」)

\begin{edaenumerate}<2>[(1)\ ]%% 小問は，edaenumerateで
\item\ $
\left(\begin{array}{@{\;}rr@{\;}}2&3\\ 1&2\end{array}\right)X=
\left(\begin{array}{@{\;}rr@{\;}}1&1\\ 0&2\end{array}\right)$
%
\item\ $X
\left(\begin{array}{@{\;}rr@{\;}}2&3\\ 1&2\end{array}\right)=
\left(\begin{array}{@{\;}rr@{\;}}1&1\\ 0&2\end{array}\right)$
%
\item\ $
\left(\begin{array}{@{\;}rr@{\;}}3&1\\ 4&2\end{array}\right)X=
\left(\begin{array}{@{\;}rr@{\;}}1&2\\ 2&4\end{array}\right)$
%
\item\ $X
\left(\begin{array}{@{\;}rr@{\;}}3&1\\ 4&2\end{array}\right)=
\left(\begin{array}{@{\;}rr@{\;}}1&2\\ 2&4\end{array}\right)$
\end{edaenumerate}
\end{\ctoi}
%%
\begin{\ckai}(解答)\par\vspace{-2mm}
\small \noindent
(1), (2) 共通に，
$\left(\begin{array}{@{\;}rr@{\;}}2&3\\ 1&2\end{array}\right)^{-1}=$
$\left(\begin{array}{@{}rr@{\;}}2&-3\\ -1&2\end{array}\right)$
\ を利用し，\par \noindent
(3), (4) 共通に，
$\left(\begin{array}{@{\;}rr@{\;}}3&1\\ 4&2\end{array}\right)^{-1}=$
$\bunsuu{1}{2}
\left(\begin{array}{@{}rr@{\;}}2&-1\\ -4&3\end{array}\right)$
\ を利用する。\par
\begin{edaenumerate}<1>[(1)\ ]%% 小問は，edaenumerateで
\item\ $X=
\left(\begin{array}{@{}rr@{\;}}2&-3\\ -1&2\end{array}\right)
\left(\begin{array}{@{\;}rr@{\;}}1&1\\ 0&2\end{array}\right)=
\left(\begin{array}{@{}rr@{\;}}2&-4\\ -1&3\end{array}\right)$
%
\item\ $X=
\left(\begin{array}{@{\;}rr@{\;}}1&1\\ 0&2\end{array}\right)
\left(\begin{array}{@{}rr@{\;}}2&-3\\ -1&2\end{array}\right)=
\left(\begin{array}{@{}rr@{\;}}1&-1\\ -2&4\end{array}\right)$
%
\item\ $X=
\bunsuu{1}{2}
\left(\begin{array}{@{}rr@{\;}}2&-1\\ -4&3\end{array}\right)
\left(\begin{array}{@{\;}rr@{\;}}1&2\\ 2&4\end{array}\right)=
\left(\begin{array}{@{}c@{\qquad}r@{\;}}0&0\\ 1&2\end{array}\right)$
%
\item\ $X=
\left(\begin{array}{@{\;}rr@{\;}}1&2\\ 2&4\end{array}\right)
\left\{
\bunsuu{1}{2}
\left(\begin{array}{@{}rr@{\;}}2&-1\\ -4&3\end{array}\right)
\right\}$
$=
\left(\begin{array}{@{}rr@{\;}}-3&\bunsuu{5}{2}\\ -6&5\end{array}\right)$
\end{edaenumerate}
\end{\ckai}

\hspace{-6mm}\framebox[3ex]{\rule{0ex}{.7em}$3$}\quad\\[-5mm]%% 大問の番号
\hspace{5mm}\begin{\ctoi}
$\left(\begin{array}{@{\;}rr@{\;}}1&3\\ 2&7\end{array}\right)
\left(\begin{array}{@{\;}r@{\;}}x\\ y\end{array}\right)=
\left(\begin{array}{@{\;}r@{\;}}4\\ 6\end{array}\right)$
をみたす列ベクトル
$\left(\begin{array}{@{\;}r@{\;}}x\\ y\end{array}\right)$
を求めよ。\\
(授業中の課題「逆行列の利用（2次）」)
\end{\ctoi}
%
\begin{\ckai}(解答)\par
\small \noindent
$\left(\begin{array}{@{\;}rr@{\;}}1&3\\ 2&7\end{array}\right)^{-1}=
\left(\begin{array}{@{}rr@{\;}}7&-3\\ -2&1\end{array}\right)$より
$\left(\begin{array}{@{\;}r@{\;}}x\\ y\end{array}\right)=
\left(\begin{array}{@{}rr@{\;}}7&-3\\ -2&1\end{array}\right)
\left(\begin{array}{@{\;}r@{\;}}4\\ 6\end{array}\right)=
\left(\begin{array}{@{}r@{\;}}10\\ -2\end{array}\right)$
\end{\ckai}
